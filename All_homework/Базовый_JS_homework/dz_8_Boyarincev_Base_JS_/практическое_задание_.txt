function b(x, y, a) {
  arguments[2] = 10;

}
b(1, 2, 3);

выведет 10 так как arguments - это массив, а 2 - это индекс аргумента массива 
от 0 до 2 это 3 позиция.

function a() {
  alert(this);
}
a.call(null);


выведет [object window], так как если this который мы вызываем 
через a.call равен undefined или null, то браузер будет ссылаться 
на глобальный объект window, а через alert(this) пытается преобразовать 
переданный объект this в строку а для глоб. объекта window строковый вид -
это [object Window].
